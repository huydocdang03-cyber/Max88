<script>
const socket=io();

function login(){
socket.emit("login",{username:username.value,password:password.value});
}

socket.on("login_success",(d)=>{
login.style.display="none";
game.style.display="block";

if(d.role==="admin") adminPanel.style.display="block";
if(d.role==="banker") bankerPanel.style.display="block";
if(d.role==="dealer") dealerPanel.style.display="block";
});

socket.on("users",(u)=>{
let html="";
for(let id in u){
html+=u[id].username+
` <button onclick="setDealer('${u[id].username}')">Chọn</button><br>`;
}
userList.innerHTML=html;

let me=u[socket.id];
if(me) money.innerText="💰 "+me.money;
});

socket.on("dealer",(d)=>{
dealer.innerText="👑 Dealer: "+d;
});

socket.on("countdown",(t)=>{
timer.innerText="⏱️ "+t;
});

socket.on("dice_result",(r)=>{
dice.innerText=r;
});

socket.on("admin_result",(r)=>{
if(r) preview.innerText="🎯 "+r.join(" - ");
});

function bet(item){
let amount=100;
socket.emit("bet",{item,amount});
}

function setDealer(n){socket.emit("set_dealer",n);}
function setResult(){socket.emit("admin_set",adminResult.value);}
function add(){socket.emit("add_chip",{user:user.value,chip:+chip.value});}
function sub(){socket.emit("sub_chip",{user:user.value,chip:+chip.value});}
function setLimit(){socket.emit("set_limit",{min:+min.value,max:+max.value});}
function openResult(){socket.emit("open_result");}
</script>
